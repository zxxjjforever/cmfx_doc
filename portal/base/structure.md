# 主程序结构
```
|--Portal
    |--Common
        |--funtion.php                //应用函数库
    |--Conf
        |--config.php                // 应用配置文件
    |--Controller
        |--ArticleController.class.php   //文章内页控制器
        |--ListController.class.php      //文章列表控制器
        |--IndexController.class.php     //应用首页控制器
        |--PageController.class.php    //页面控件器
    |--Lang                            //多语言文件
    |--Menu                          //1.0.4版本目录，默认为空用来存放备份的此应用的后台菜单
    |--nav.php                    //1.0.4新加，用于开发者控制返回前台菜单的文件
```
